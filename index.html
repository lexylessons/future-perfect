    <script>

        function showTab(tabId) {

            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

            document.querySelectorAll('.nav-tab').forEach(tab => tab.classList.remove('active'));

            

            document.getElementById(tabId).classList.add('active');

            event.target.classList.add('active');

            window.scrollTo(0,0);

        }


        function checkForm(el, correct) {

            if (el.value === correct) {

                el.classList.add('border-green-500', 'bg-green-100');

            } else {

                el.classList.add('border-red-500', 'bg-red-100');

            }

        }


        function checkDiscovery() {

            const q1 = document.querySelector('input[name="q1"]:checked')?.value;

            const q2 = document.querySelector('input[name="q2"]:checked')?.value;

            

            if (q1 === 'correct' && q2 === 'correct') {

                alert("Excellent! You've grasped the meaning of both tenses.");

            } else {

                alert("Review Step 1 again. Look at the difference between 'completion' and 'duration'.");

            }

        }


        function checkPractice() {

            // Check Exercise A

            document.querySelectorAll('.ex-row').forEach(row => {

                const input = row.querySelector('input');

                const ans = row.dataset.ans.toLowerCase();

                const user = input.value.toLowerCase().trim();

                

                if (user === ans) {

                    input.className = "w-full p-3 border rounded-lg correct outline-none";

                } else {

                    input.className = "w-full p-3 border rounded-lg incorrect outline-none";

                    input.value += ` (Correct: ${ans})`;

                }

            });


            // Check Exercise B

            document.querySelectorAll('[data-fix]').forEach(input => {

                const ans = input.dataset.fix.toLowerCase();

                const user = input.value.toLowerCase().trim();

                

                if (user === ans) {

                    input.className = "p-2 border rounded w-1/3 correct";

                } else {

                    input.className = "p-2 border rounded w-1/3 incorrect";

                    input.value = ans;

                }

            });

        }
